\begin{homeworkProblem}
    Consider the nonlinear system
    \[
        \begin{split}
            5x^2 - x^2 &= 0\\
            x_2 - (\sin(x_1) + \cos(x_2))/4 &= 0
        \end{split}
    \]

    \begin{enumerate}[a)]
        \item Find a function \(G(\vec{x})\) and a set \(D\) in \(\mathbb{R}^2\) such that \(G\) has a unique fixed point in \(D\).
        \item Estimate the number of iterations required to approximate the exact solution within \(10^{-5}\) in the \(||\cdot||_\infty\) norm, give nany initial guess in \(D\).
    \end{enumerate}
\end{homeworkProblem}

\begin{homeworkProblem}
    Use two iterations of Newton's method with initial guess \(vec{0}\) to approximate the solution to
    \[
        \begin{split}
            3x - \cos(yz) - 1/2 &= 0\\
            4x^2 - 625y^2 + 2y - 1 &= 0\\
            e^{-xy} + 20z + \frac{10\pi - 3}{3} &= 0
        \end{split}
    \]
\end{homeworkProblem}
